Ejercicio 12 - Loop Control

Uso de break

for (var i = 1; i <= 10; i++){
    if(i == 5){
        break;
    }
    document.write("Valor: " + i + "
");
}

Uso de continue

for (var i = 1; i <= 5; i++){
    if(i == 3){
        continue;
    }
    document.write("Número: " + i + "
");
}

Uso de etiquetas (labels)

etiqueta1:
for (var i = 1; i <= 3; i++){
    for (var j = 1; j <= 3; j++){
        if(i == 2 && j == 2){
            break etiqueta1;
        }
        document.write("i = " + i + ", j = " + j + "
");
    }
}
