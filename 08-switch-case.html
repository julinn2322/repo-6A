Ejercicio 08 - Switch Case

Switch básico

var dia = 3;
var nombre;

switch (dia) {
    case 1: nombre = "Lunes"; break;
    case 2: nombre = "Martes"; break;
    case 3: nombre = "Miércoles"; break;
    case 4: nombre = "Jueves"; break;
    case 5: nombre = "Viernes"; break;
    default: nombre = "Fin de semana";
}

document.write("Hoy es " + nombre);

Switch con múltiples casos

var fruta = "manzana";
var mensaje;

switch (fruta) {
   case "manzana":
   case "pera":
   case "durazno":
       mensaje = "Es una fruta comestible";
       break;
   default:
       mensaje = "No es una fruta del grupo";
}

document.write(mensaje);
