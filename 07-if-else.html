Ejercicio 07 - If / Else

IF Simple

var edad = 20;

if(edad > 18){
   document.write("Es mayor de edad");
}

IF / ELSE

var edad = 16;

if(edad >= 18){
   document.write("Mayor de edad");
} else {
   document.write("Menor de edad");
}

IF / ELSE IF / ELSE

var nota = 8;

if(nota == 10){
    document.write("Excelente");
} else if (nota >= 7){
    document.write("Muy bien");
} else if (nota >= 4){
    document.write("Aprobado");
} else {
    document.write("Reprobado");
}
